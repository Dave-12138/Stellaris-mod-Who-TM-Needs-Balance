# 拆巨构
wtmnb_edict_4 = {
	icon = GFX_nullobject
	length = 1
	resources = {
		category = edicts
		upkeep = {
			energy = 1
		}
	}
	effect = {
		hidden_effect = {
			# end_event_chain = wtmnb_remove_megastructure_chain
			set_variable = {
				which = wtmnb_rmmega_chain_count
				value = 0
			}
		}
		begin_event_chain = {
			event_chain = wtmnb_remove_megastructure_chain
		}
		every_owned_ship = {
			limit = {
				is_ship_class = shipclass_constructor
			}
			solar_system = {
				every_system_megastructure = {
					enable_special_project = {
						name = WTMNB_REMOVE_MEGASTRUCTURE
						location = this
						owner = root
					}
					hidden_effect = {
						root = {
							change_variable = {
								which = wtmnb_rmmega_chain_count
								value = 1
							}
						}
					}
				}
			}
		}
		if = {
			limit = {
				check_variable = {
					which = wtmnb_rmmega_chain_count
					value < 1
				}
			}
			end_event_chain = wtmnb_remove_megastructure_chain
		}
	}
	potential = {
		is_ai = no
	}
	allow = {
		is_ai = no
		NOT = {
			has_event_chain = wtmnb_remove_megastructure_chain
		}
	}
}
