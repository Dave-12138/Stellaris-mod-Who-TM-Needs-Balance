# 拆巨构
wtmnb_edict_4 = {
	icon = GFX_resource_engineering
	length = 1
	resources = {
		category = edicts
		upkeep = {
			energy = 1
		}
	}
	effect = {
		hidden_effect = {
			# end_event_chain = wtmnb_remove_megastructure_chain
			set_variable = {
				which = wtmnb_rmmega_chain_count
				value = 0
			}
		}
		begin_event_chain = {
			event_chain = wtmnb_remove_megastructure_chain
		}
		every_owned_ship = {
			limit = {
				is_ship_class = shipclass_constructor
			}
			solar_system = {
				set_star_flag = wtmnb_rmmega_system
			}
		}
		every_system = {
			limit = {
				has_star_flag = wtmnb_rmmega_system
			}
			remove_star_flag = wtmnb_rmmega_system
			every_system_megastructure = {
				enable_special_project = {
					name = WTMNB_REMOVE_MEGASTRUCTURE
					location = this
					owner = root
				}
				hidden_effect = {
					root = {
						change_variable = {
							which = wtmnb_rmmega_chain_count
							value = 1
						}
					}
				}
			}
		}
		if = {
			limit = {
				check_variable = {
					which = wtmnb_rmmega_chain_count
					value < 1
				}
			}
			end_event_chain = wtmnb_remove_megastructure_chain
		}
	}
	potential = {
		has_country_flag = wtmnb_order
	}
	allow = {
		is_ai = no
		NOT = {
			has_event_chain = wtmnb_remove_megastructure_chain
		}
	}
}

wtmnb_edict_create_ship = {
	icon = GFX_ship_counter_32
	length = -1
	resources = {
		category = edicts
		upkeep = {
			alloys = 1
		}
	}
	effect = {
		# create_ship_design = {
		# 	design = "DESIGN_wtmnb_fortress"
		# }
		# add_ship_design = last_created_design
		capital_scope = {
			create_fleet = {
				settings = {
					can_upgrade = no
					spawn_debris = no
					can_change_composition = no
					uses_naval_capacity = no
				}
				effect = {
					set_owner = root
					create_ship = {
						name = wtmnb_fortress
						design = "DESIGN_wtmnb_fortress"
						upgradable = no
					}
					set_location = prev
					set_fleet_stance = passive
					set_aggro_range_measure_from = self
					set_aggro_range = 350
				}
			}
		}
	}
	potential = {
		has_country_flag = wtmnb_order
		NOT = {
			any_owned_ship = {
				is_ship_size = wtmnb_fortress
			}
		}
		is_ai = no
	}
	allow = {
		is_ai = no
	}
}
