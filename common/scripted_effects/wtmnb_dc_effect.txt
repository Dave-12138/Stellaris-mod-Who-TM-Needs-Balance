show_dc_r_desc = {
	if = {
		limit = {
			has_ruler_trait = leader_trait_ruler_wtmnb_dc_r
		}
		custom_tooltip = leader_trait_ruler_wtmnb_dc_r_added
	}
	else_if = {
		limit = {
			has_ruler_trait = leader_trait_ruler_wtmnb_dc_r2
		}
		custom_tooltip = leader_trait_ruler_wtmnb_dc_r2_added
	}
	if = {
		limit = {
			has_ruler_trait = leader_trait_ruler_wtmnb_dc_r3
		}
		custom_tooltip = leader_trait_ruler_wtmnb_dc_r3_added
	}
}

wtmnb_planet_species_to_suc = {
	every_owned_species = {
		limit = {
			NOT = {
				is_species_class = "SUC"
			}
		}
		create_species = {
			class = "SUC"
			is_mod = yes
			name = this
			plural = this
			portrait = this
			traits = this
			homeworld = prev
		}
		modify_species = {
			species = last_created_species
			remove_trait = trait_DC_ethic_asceticism
			add_trait = trait_DC_ethic_carnalism
			add_traits_at_start_of_list = yes
		}
		modify_species = {
			species = last_created_species
			remove_trait = trait_DC_ethic_parasite_mind
			add_trait = trait_sensitive_vagina
			add_traits_at_start_of_list = yes
		}
		modify_species = {
			species = last_created_species
			remove_trait = trait_DC_ethic_lust_skynet
			add_trait = trait_big_breasts
			add_traits_at_start_of_list = yes
		}
		modify_species = {
			species = last_created_species
			add_trait = trait_coquettish
			add_traits_at_start_of_list = yes
		}
		wtmnb_normal_species_fix = yes
		prev = {
			every_owned_pop = {
				limit = {
					is_same_species = prevprev
				}
				change_species = last_created_species
			}
		}
	}
}
